2. Draw plots of model performances versus the hyperparameter¶

In [14]:
from sklearn.linear_model import Ridge
lst_alpha = np.arange(0, 2.1, 0.1)
lst_train_rmse = []
lst_test_rmse = []

for alpha in lst_alpha :
    reg = Ridge(alpha=alpha)
    reg.fit(X_train, y_train)
    y_train_hat = reg.predict(X_train)
    y_test_hat = reg.predict(X_test)
    lst_train_rmse.append(mean_squared_error(y_train, y_train_hat)**0.5)
    lst_test_rmse.append(mean_squared_error(y_test, y_test_hat)**0.5)
In [15]:
plt.plot(lst_alpha, lst_train_rmse, label = 'train error')
plt.plot(lst_alpha, lst_test_rmse, label = 'test error')

plt.legend()
plt.xlabel('hyperparameter lambda')
plt.ylabel('RMSE')
plt.xlim(0,2)
